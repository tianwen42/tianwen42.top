    namespce

    定义和使用

    同一命名空间不可重名，可以是分散的

    #include <iostream>
using namespace std;

namespace A{
    int i=1,j=2;
}

namespace A{
    int k=3;
}
namespace B{
    int i=4,j=5,k=6;
}

int foo(){
    std::cout<<"namespace A:"<<A::i<<A::j<<A::k<<std::endl;
    std::cout<<"namespace B:"<<B::i<<B::j<<B::k<<std::endl;
    return A::i;
}

    首先区分引用变量与指针的区别

    int * ptr;    // 声明指针变量ptr
ptr = &var;   // 先使用 & 运算符获取变量var的地址，再把该地址赋值给指针变量ptr
int ** pptr;  // 声明二级指针变量pptr
pptr = &ptr;  // 先使用 & 运算符获取变量ptr的地址，再把该地址赋值给二级指针变量pptr
int & ref1 = var;   // 声明引用变量ref1, ref1是变量var的别名(引用必须在创建时被初始化)
int & ref2 = *ptr

    &*p //指针的地址& *p=p=&a

    *&var //var的数值 *&var=var= *p

    指向const对象的指针

    无法改变指向对象的值，但可以改变指向

    const int num=1;
const int *p1;	        //不允许通过指针来改变对象->*p是只读的
p1=&num;
*p1=2;                  //错误
int *p2=&num;           //错误,p2只能指向的是非const对象
const int *p3=&num;     //正确

    作用

    int foo(const int *p){
    //process
    return 0;
}
int main(){
    int a=0;
    foo(&a);    //无法改变a值
    return 0;
}

    指向非const对象的const指针

    无法改变指向，只能指向唯一对象，但可以间接引用改变对象值

    int main(){
    int a=1,b=2;
    int * const pc=&a;
    pc = &b;    //错误,pc只读,不能赋值
    pc=pc;      //错误,pc只读
    pc++;       //错误,pc只读,不能运算
    *pc=100;    //正确
    return 0;
}

    指向const对象的const指针

    无法改变指向，也无法改变值

    const int a=1,b=2;
const int * const pc=&a;
*pc=100;	//错误，指向const对象的const指针无法修改值
