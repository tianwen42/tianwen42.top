def cal_view(request, m, op, n):
    if op not in ['add', 'sub', 'nul']:
        return HttpResponse('Your op is not correct')
    result = 0
    if op == 'add':
        result = m + n
    elif op == 'sub':
        result = m - n
    elif op == 'nul':
        result = n * m
    return HttpResponse('结果为{}'.format(result))
